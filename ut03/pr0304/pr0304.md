## Escribe un script que tome dos parámetros: un archivo y una palabra clave. El script debe imprimir todas las líneas del archivo que contengan la palabra clave.

```
#!/bin/bash

# 1. Validar que el usuario envió exactamente 2 argumentos
if [ $# -ne 2 ]; then
    echo "Uso: $0 <archivo> <palabra_clave>"
    exit 1
fi

# 2. Asignar nombres legibles a los parámetros
ARCHIVO=$1
PALABRA=$2

# 3. Verificar si el archivo realmente existe
if [ ! -f "$ARCHIVO" ]; then
    echo "Error: El archivo '$ARCHIVO' no existe."
    exit 1
fi

# 4. Buscar la palabra
echo "Buscando '$PALABRA' en $ARCHIVO..."
echo "--------------------------------"
grep --color=always "$PALABRA" "$ARCHIVO"
```

## Crea un script que reciba un archivo de texto como parámetro y cuente la cantidad de palabras. Muestra el resultado en pantalla.

```
#!/bin/bash

# 1. Verificar que se haya pasado un argumento
if [ $# -ne 1 ]; then
    echo "Uso: $0 <archivo>"
    exit 1
fi

# 2. Asignar el parámetro a una variable
ARCHIVO=$1

# 3. Validar que el archivo exista y sea un archivo regular
if [ ! -f "$ARCHIVO" ]; then
    echo "Error: El archivo '$ARCHIVO' no existe o no es un archivo válido."
    exit 1
fi

# 4. Contar las palabras
# Usamos 'wc -w' para obtener solo el conteo de palabras
TOTAL=$(wc -w < "$ARCHIVO")

echo "El archivo '$ARCHIVO' tiene $TOTAL palabras."
```

## Realiza un script que reciba un archivo de texto con números (uno por línea) y calcule la suma total. El resultado debe mostrarse al final.

```
#!/bin/bash

# 1. Validar que se pasó el archivo
if [ $# -ne 1 ]; then
    echo "Uso: $0 <archivo_con_numeros>"
    exit 1
fi

ARCHIVO=$1
SUMA=0

# 2. Verificar existencia del archivo
if [ ! -f "$ARCHIVO" ]; then
    echo "Error: El archivo '$ARCHIVO' no existe."
    exit 1
fi

# 3. Bucle para leer el archivo línea por línea
while read -r linea || [[ -n "$linea" ]]; do
    # Sumar solo si la línea no está vacía y es un número
    if [[ "$linea" =~ ^-?[0-9]+$ ]]; then
        SUMA=$((SUMA + linea))
    fi
done < "$ARCHIVO"

# 4. Mostrar resultado
echo "La suma total de los números en '$ARCHIVO' es: $SUMA"
```

## Supón que tienes un archivo CSV donde cada línea contiene los campos: nif, nombre, apellidos, ciclo, módulo, nota, Crea un script al que se le pase como parámetro un número de NIF y busque las líneas que correspondan a dicho NIF, mostrando para cada línea un mensaje de la forma: El alumno {nombre apellidos} tiene una calificación de {nota} puntos en el módulo {módulo}.

10100100X, victor, gonzález rodríguez, asir, aso, 6
12345678A, pepe, fernández fernández, asir, aso, 8
10100100X, victor, gonzález rodríguez, asir, iso, 5

```
#!/bin/bash

# 1. Validar que se pasaron exactamente 2 argumentos: el CSV y el NIF
if [ $# -ne 2 ]; then
    echo "Uso: $0 <archivo.csv> <NIF>"
    exit 1
fi

ARCHIVO=$1
NIF_BUSCADO=$2
ENCONTRADO=false

# 2. Verificar si el archivo existe
if [ ! -f "$ARCHIVO" ]; then
    echo "Error: El archivo '$ARCHIVO' no existe."
    exit 1
fi

# 3. Procesar el archivo línea por línea
while IFS=',' read -r nif nombre apellidos ciclo modulo nota
do
    # Limpiar posibles espacios en blanco alrededor de las variables
    nif=$(echo "$nif" | xargs)
    
    # Comprobar si el NIF de la línea coincide con el buscado
    if [ "$nif" == "$NIF_BUSCADO" ]; then
        # Limpiar el resto de variables para que el mensaje se vea bien
        nombre=$(echo "$nombre" | xargs)
        apellidos=$(echo "$apellidos" | xargs)
        modulo=$(echo "$modulo" | xargs)
        nota=$(echo "$nota" | xargs)

        echo "El alumno $nombre $apellidos tiene una calificación de $nota puntos en el módulo $modulo."
        ENCONTRADO=true
    fi
done < "$ARCHIVO"

# 4. Mensaje opcional si no se encuentra nada
if [ "$ENCONTRADO" = false ]; then
    echo "No se encontraron registros para el NIF: $NIF_BUSCADO"
fi
```

## Escribe un script que reciba un archivo y un número N como parámetros. El script debe imprimir todas las líneas del archivo que tengan más de N caracteres.

```
#!/bin/bash

# 1. Validar que se pasaron los 2 parámetros requeridos
if [ $# -ne 2 ]; then
    echo "Uso: $0 <archivo> <número_N>"
    exit 1
fi

ARCHIVO=$1
N=$2

# 2. Verificar que el archivo existe
if [ ! -f "$ARCHIVO" ]; then
    echo "Error: El archivo '$ARCHIVO' no existe."
    exit 1
fi

# 3. Verificar que N sea un número entero
if ! [[ "$N" =~ ^[0-9]+$ ]]; then
    echo "Error: '$N' no es un número válido."
    exit 1
fi

# 4. Leer el archivo y comparar la longitud de cada línea
while IFS= read -r linea
do
    # ${#linea} devuelve la cantidad de caracteres de la variable
    if [ ${#linea} -gt "$N" ]; then
        echo "$linea"
    fi
done < "$ARCHIVO"
```

## Crea un script que tome varios archivos como parámetros y cuente las líneas, palabras y caracteres de cada archivo, mostrando un pequeño informe de cada uno en pantalla.

```
#!/bin/bash

# 1. Verificar si se pasó al menos un archivo
if [ $# -lt 1 ]; then
    echo "Uso: $0 archivo1 [archivo2 ... archivoN]"
    exit 1
fi

echo "=========================================="
echo "      INFORME DE ESTADÍSTICAS"
echo "=========================================="

# 2. Recorrer todos los archivos pasados como argumentos
for ARCHIVO in "$@"
do
    if [ -f "$ARCHIVO" ]; then
        # Extraer datos usando wc
        LINEAS=$(wc -l < "$ARCHIVO")
        PALABRAS=$(wc -w < "$ARCHIVO")
        CARACTERES=$(wc -m < "$ARCHIVO")

        echo "Archivo: $ARCHIVO"
        echo "  - Líneas:     $LINEAS"
        echo "  - Palabras:   $PALABRAS"
        echo "  - Caracteres: $CARACTERES"
        echo "------------------------------------------"
    else
        echo "Error: '$ARCHIVO' no existe o no es un archivo válido."
        echo "------------------------------------------"
    fi
done
```
